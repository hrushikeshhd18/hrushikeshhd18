<script>
  function getRandomInt(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
  }

  function createParticle() {
    var particle = document.createElement('div');
    particle.className = 'particle';
    particle.style.left = getRandomInt(0, window.innerWidth) + 'px';
    particle.style.top = getRandomInt(0, window.innerHeight) + 'px';
    particle.style.animationDuration = getRandomInt(5, 15) + 's';
    particle.style.animationDelay = getRandomInt(0, 5) + 's';

    document.getElementById('particles-js').appendChild(particle);
  }

  for (var i = 0; i < 50; i++) {
    createParticle();
  }
</script>
